import React, { useState } from 'react';

export default function App() {
  const [name, setName] = useState("");

  return (
    <div style={{ padding: '20px' }}>
      <input 
        type="text" 
        placeholder="Enter your name..." 
        onChange={(e) => setName(e.target.value)} 
      />
      <p>Hello, {name || "Stranger"}!</p>
    </div>
  );
}
