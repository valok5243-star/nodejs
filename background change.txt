background change

import React, { useState } from 'react';

export default function App() {
  const [isBlue, setIsBlue] = useState(true);

  return (
    <div 
      style={{ 
        width: '200px', 
        height: '200px', 
        backgroundColor: isBlue ? 'skyblue' : 'tomato',
        display: 'flex',
        alignItems: 'center',
        justifyContent: 'center',
        cursor: 'pointer'
      }}
      onClick={() => setIsBlue(!isBlue)}
    >
      Click to change color
    </div>
  );
}
